# first_api
1. creation repo
    '''
    mkdir simple_flask_app
    cd simple_flask_app
    '''

2. install & activate virtual environment
    - anaconda
        '''
        conda create -n name python=3.xx
        conda activate name
        '''

3. install flask
    '''
    conda install Flask
    '''

4. make app.py file

5. run app
    '''
    python app.py
    '''
